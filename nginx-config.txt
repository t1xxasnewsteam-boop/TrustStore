# Добавь эти строки в конфигурацию Nginx для TrustStore

location / {
    # Попытка найти файл, если не найден - попробовать с .html
    try_files $uri $uri.html $uri/ =404;
    
    # Убираем .html из URL (редирект)
    if ($request_uri ~ ^/(.*)\.html$) {
        return 301 /$1;
    }
}

# Для корневой страницы
location = / {
    try_files /main.html /index.html =404;
}

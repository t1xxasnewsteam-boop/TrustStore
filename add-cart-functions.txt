// VPN
function addProductToCart() {
    const product = {
        name: 'Личный VPN',
        price: parseInt(document.getElementById('selected-price').textContent.replace(' ₽', '').replace(/\s/g, '')),
        duration: selectedDuration || '1 месяц',
        category: 'Безопасность',
        image: 'vpn-image.png'
    };
    addToCart(product);
}
